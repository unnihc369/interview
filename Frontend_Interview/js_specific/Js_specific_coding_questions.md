## 1. First Non-Repeating Character

**Problem:** Return the first character that does not repeat.

```js
function firstNonRepeating(str) {
  const freq = {};
  for (const ch of str) freq[ch] = (freq[ch] || 0) + 1;
  for (const ch of str) if (freq[ch] === 1) return ch;
  return null;
}
// firstNonRepeating("leetcode") → "l"
```

## 2. Rotate Array Right by k

**Problem:** Rotate in-place without extra array. Use reverse trick.

```js
function rotateRight(arr, k) {
  const n = arr.length;
  if (n === 0) return arr;
  k = k % n;
  const reverse = (a, l, r) => {
    while (l < r) [a[l], a[r]] = [a[r], a[l]], l++, r--;
  };
  reverse(arr, 0, n - 1);
  reverse(arr, 0, k - 1);
  reverse(arr, k, n - 1);
  return arr;
}
// rotateRight([1,2,3,4,5], 2) → [4,5,1,2,3]
```

## 3. Longest Substring Without Repeating Characters

```js
function lengthOfLongestSubstring(s) {
  const map = new Map();
  let max = 0, left = 0;
  for (let right = 0; right < s.length; right++) {
    if (map.has(s[right]) && map.get(s[right]) >= left) {
      left = map.get(s[right]) + 1;
    }
    map.set(s[right], right);
    max = Math.max(max, right - left + 1);
  }
  return max;
}
// lengthOfLongestSubstring("abcabcbb") → 3
```

## 4. Move Zeros to End

```js
function moveZeros(arr) {
  let insert = 0;
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] !== 0) arr[insert++] = arr[i];
  }
  while (insert < arr.length) arr[insert++] = 0;
  return arr;
}
```

## 5. Anagram Check

```js
function isAnagram(s, t) {
  if (s.length !== t.length) return false;
  const count = {};
  for (const ch of s) count[ch] = (count[ch] || 0) + 1;
  for (const ch of t) {
    if (!count[ch]) return false;
    count[ch]--;
  }
  return true;
}
// isAnagram("listen", "silent") → true
```

```js
function secondHighestIndex(arr) {
  let first = -Infinity, second = -Infinity, firstIdx = -1, secondIdx = -1;
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] > first) {
      second = first; secondIdx = firstIdx;
      first = arr[i]; firstIdx = i;
    } else if (arr[i] > second && arr[i] !== first) {
      second = arr[i]; secondIdx = i;
    }
  }
  return secondIdx;
}
```

```js
function maxSubArray(nums) {
```

## 10. Unique Pairs Sum to Target

```js
function pairSum(arr, target) {
  const seen = new Set();
  const pairs = [];
  const used = new Set();
  for (const num of arr) {
    const complement = target - num;
    const key = [Math.min(num, complement), Math.max(num, complement)].join(",");
    if (seen.has(complement) && !used.has(key)) {
      pairs.push([complement, num]);
      used.add(key);
    }
    seen.add(num);
  }
  return pairs;
}
// pairSum([1,2,3,4,5], 6) → [[1,5],[2,4]]
```

## 17. Longest Palindromic Substring

```js
function longestPalindrome(s) {
  function expand(l, r) {
    while (l >= 0 && r < s.length && s[l] === s[r]) l--, r++;
    return s.slice(l + 1, r);
  }
  let longest = "";
  for (let i = 0; i < s.length; i++) {
    const odd = expand(i, i);
    const even = expand(i, i + 1);
    const best = odd.length >= even.length ? odd : even;
    if (best.length > longest.length) longest = best;
  }
  return longest;
}
// longestPalindrome("babad") → "bab" or "aba"
```

## 19. Backspace String Compare

```js
function backspaceCompare(s, t) {
  function build(str) {
    const stack = [];
    for (const ch of str) {
      if (ch === "#") stack.pop();
      else stack.push(ch);
    }
    return stack.join("");
  }
  return build(s) === build(t);
}
```

## 20. Merge Two Sorted Arrays

```js
function mergeSorted(a, b) {
  const result = [];
  let i = 0, j = 0;
  while (i < a.length && j < b.length) {
    if (a[i] <= b[j]) result.push(a[i++]);
    else result.push(b[j++]);
  }
  return result.concat(a.slice(i), b.slice(j));
}
```

## 23. All Permutations

```js
function permutations(str) {
  const result = [];
  function backtrack(path, used) {
    if (path.length === str.length) {
      result.push(path.join(""));
      return;
    }
    for (let i = 0; i < str.length; i++) {
      if (used[i]) continue;
      used[i] = true;
      path.push(str[i]);
      backtrack(path, used);
      path.pop();
      used[i] = false;
    }
  }
  backtrack([], Array(str.length).fill(false));
  return result;
}
// permutations("ab") → ["ab","ba"]
```

## 24. Product Except Self

```js
function productExceptSelf(nums) {
  const n = nums.length;
  const result = Array(n).fill(1);
  let prefix = 1;
  for (let i = 0; i < n; i++) {
    result[i] = prefix;
    prefix *= nums[i];
  }
  let suffix = 1;
  for (let i = n - 1; i >= 0; i--) {
    result[i] *= suffix;
    suffix *= nums[i];
  }
  return result;
}
// productExceptSelf([1,2,3,4]) → [24,12,8,6]
```

## 25. Same Pattern

```js
function samePattern(pattern, str) {
  const words = str.split(" ");
  if (pattern.length !== words.length) return false;
  const pToS = {}, sToP = {};
  for (let i = 0; i < pattern.length; i++) {
    const p = pattern[i], w = words[i];
    if (pToS[p] !== undefined && pToS[p] !== w) return false;
    if (sToP[w] !== undefined && sToP[w] !== p) return false;

    sToP[w] = p;
  }

}
```

## 27. Balanced Brackets

```js
function isBalanced(s) {
  const stack = [];
  const map = { ")": "(", "]": "[", "}": "{" };
  for (const ch of s) {
    if ("([{".includes(ch)) stack.push(ch);
    else if (")]}".includes(ch)) {
      if (stack.pop() !== map[ch]) return false;
    }
  }
  return stack.length === 0;
}
// isBalanced("({[]})") → true
```

## 28. All Subsets

```js
function subsets(nums) {

    result.push([...path]);
```

## 30. Subarrays with Sum = Target

```js
function subarraySum(nums, target) {
  const prefix = new Map([[0, 1]]);
  let sum = 0, count = 0;
  for (const num of nums) {
    sum += num;
    if (prefix.has(sum - target)) count += prefix.get(sum - target);
    prefix.set(sum, (prefix.get(sum) || 0) + 1);
  }
  return count;
}
// subarraySum([1,1,1], 2) → 2
```

## 32. Triplets Sum to Zero

```js
function threeSum(nums) {
  nums.sort((a, b) => a - b);
  const result = [];
  for (let i = 0; i < nums.length - 2; i++) {
    if (i > 0 && nums[i] === nums[i - 1]) continue;
    let l = i + 1, r = nums.length - 1;
    while (l < r) {
      const sum = nums[i] + nums[l] + nums[r];
      if (sum === 0) {
        result.push([nums[i], nums[l], nums[r]]);
        while (l < r && nums[l] === nums[l + 1]) l++;
        while (l < r && nums[r] === nums[r - 1]) r--;
        l++; r--;
      } else if (sum < 0) l++;
      else r--;
    }
  }
  return result;
}
// threeSum([-1,0,1,2,-1,-4]) → [[-1,-1,2],[-1,0,1]]
```

## 33. String Compression

```js
function compressString(s) {
  if (!s) return s;
  let result = "", count = 1;
  for (let i = 1; i <= s.length; i++) {
    if (s[i] === s[i - 1]) count++;
    else {
      result += s[i - 1] + (count > 1 ? count : "");
      count = 1;
    }
  }
  return result.length < s.length ? result : s;
}
// compressString("aabbb") → "a2b3"
```

## 36. Longest Increasing Subsequence

```js
function lengthOfLIS(nums) {
  const dp = [];
  for (const num of nums) {
    let lo = 0, hi = dp.length;
    while (lo < hi) {

}
```

```js
function countPairsWithDiff(arr, k) {
  const set = new Set(arr);
  let count = 0;
  for (const num of set) {
```

## 39. Roman to Integer

```js
function romanToInt(s) {
  const map = { I:1, V:5, X:10, L:50, C:100, D:500, M:1000 };
  let total = 0;
  for (let i = 0; i < s.length; i++) {
    const curr = map[s[i]], next = map[s[i + 1]];
    total += curr < next ? -curr : curr;
  }
  return total;
}
// romanToInt("MCMXCIV") → 1994
```

## 58. Kth Smallest

```js
function kthSmallest(nums, k) {
  const minHeap = [...nums];
  const heapify = (arr, n, i) => {
    let smallest = i, l = 2 * i + 1, r = 2 * i + 2;
    if (l < n && arr[l] < arr[smallest]) smallest = l;
    if (r < n && arr[r] < arr[smallest]) smallest = r;
    if (smallest !== i) {
      [arr[i], arr[smallest]] = [arr[smallest], arr[i]];
      heapify(arr, n, smallest);
    }
  };
  for (let i = Math.floor(minHeap.length / 2) - 1; i >= 0; i--) heapify(minHeap, minHeap.length, i);
  for (let i = minHeap.length - 1; i > 0; i--) {
    [minHeap[0], minHeap[i]] = [minHeap[i], minHeap[0]];
    heapify(minHeap, i, 0);

  return minHeap[k - 1];
}
```

```js
    if (a.length <= 1) return a;
    const mid = a.length >> 1;
```

```js
  return [...strs].sort((a, b) => a.length - b.length);
}
// sortByLength(["hi","hello","hey"]) → ["hi","hey","hello"]
```
